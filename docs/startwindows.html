 22class TrialManager:
 23    def __init__(self, trial_length_days=30, trial_file='dax.txt'):
 24        self.trial_length_days = trial_length_days
 25        self.trial_file = trial_file
 26
 27    def read_or_start_trial(self):
 28        if not os.path.exists(self.trial_file):
 29            with open(self.trial_file, 'w') as file:
 30                file.write(datetime.datetime.now().isoformat())
 31            crypto.encrypt_dax_file(self.trial_file)
 32            return self.trial_length_days
 33
 34        crypto.decrypt_dax_file(self.trial_file)
 35        with open(self.trial_file, 'r') as file:
 36            file_content = file.read().strip()
 37            crypto.encrypt_dax_file(self.trial_file)  # Re-encrypt after reading
 38
 39            if file_content == 'TRIAL_ENDED':
 40                return 0
 41
 42            start_date = datetime.datetime.fromisoformat(file_content)
 43            remaining = self.trial_length_days - (datetime.datetime.now() - start_date).days
 44
 45            return max(0, remaining)
 46
 47    def end_trial(self):
 48        with open(self.trial_file, 'w') as file:
 49            file.write('TRIAL_ENDED')
 50        crypto.encrypt_dax_file(self.trial_file)
 51
 52    def check_trial(self):
 53        crypto.decrypt_dax_file(self.trial_file)
 54        with open(self.trial_file, 'r') as file:
 55            trial_data = file.read().strip()
 56            crypto.encrypt_dax_file(self.trial_file)
 57        return trial_data == 'TRIAL_ENDED'
